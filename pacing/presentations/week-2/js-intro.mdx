```js
// operators: arithmetic
// PEMDAS (parenthesis, exponent, multiplication, division, addition, subtraction)
let foo = 1

foo = foo + 1 // 2

foo++ // 3 (foo = foo + 1)
foo-- // 2 (foo = foo - 1)

foo = 2 * 2 // 4
foo = (5 * 4) / 2 // => 10 (20 / 2)

foo *= 2 // 20 (basically is saying foo is whatever foo is multiplied by 20)

4**2 // 16 (4 squared - exponent)
foo**2 // 200

// all these operators work with objects and arrays too....

let obj = {
    num: 5
}
obj.num++ // 6

obj // {num: 6}

let arr = [6]
arr[0]++
arr[0] // 7

```
